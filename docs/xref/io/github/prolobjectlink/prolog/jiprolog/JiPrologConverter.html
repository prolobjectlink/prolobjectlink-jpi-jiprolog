50  	public PrologTerm toTerm(JIPTerm prologTerm) {
51  		if (prologTerm instanceof JIPAtom) {
52  			JIPAtom atom = (JIPAtom) prologTerm;
53  			if (atom.getName().equals("!")) {
54  				return new JiPrologCut(provider);
55  			} else if (atom.getName().equals("nil")) {
56  				return new JiPrologNil(provider);
57  			} else if (atom.getName().equals("true")) {
58  				return new JiPrologTrue(provider);
59  			} else if (atom.getName().equals("false")) {
60  				return new JiPrologFalse(provider);
61  			} else if (atom.getName().equals("fail")) {
62  				return new JiPrologFail(provider);
63  			} else {
64  				return new JiPrologAtom(provider, atom.getName());
65  			}
66  		} else if (prologTerm instanceof JIPNumber) {
67  			JIPNumber number = (JIPNumber) prologTerm;
68  			if (number.isInteger()) {
69  				return new JiPrologInteger(provider, number.getDoubleValue());
70  			} else {
71  				return new JiPrologDouble(provider, number.getDoubleValue());
72  			}
73  		} else if (prologTerm instanceof JIPVariable) {
74  			String name = ((JIPVariable) prologTerm).getName();
75  			PrologVariable variable = sharedVariables.get(name);
76  			if (variable == null) {
77  				variable = new JiPrologVariable(provider, name);
78  				sharedVariables.put(variable.getName(), variable);
79  			}
80  			return variable;
81  		} else if (prologTerm instanceof JIPList) {
82  			JIPList list = (JIPList) prologTerm;
83  			if (!list.isNIL()) {
84  				JIPTerm[] arguments = new JIPTerm[list.length()];
85  				for (int i = 0; i < arguments.length; i++) {
86  					try {
87  						arguments[i] = list.getNth(i + 1);
88  					} catch (NullPointerException e) {
89  						// is raised because the current list has [T|[]] format
90  						arguments = Arrays.copyOf(arguments, arguments.length - 1);
91  					}
92  				}
93  				return new JiPrologList(provider, arguments);
94  			}
95  			return new JiPrologEmpty(provider);
96  		} else if (prologTerm instanceof JIPFunctor) {
97  			JIPFunctor structure = (JIPFunctor) prologTerm;
98  			String functor = structure.getName();
99  			if (!new JiPrologOperatorSet().currentOp(functor)) {
100 				JIPTerm[] arguments = new JIPTerm[structure.getArity()];
101 				for (int i = 0; i < arguments.length; i++) {
102 					arguments[i] = structure.getTerm(i + 1);
103 				}
104 				return new JiPrologStructure(provider, functor, arguments);
105 			} else if (structure.getArity() == 2) {
106 				JIPTerm left = structure.getTerm(1);
107 				JIPTerm right = structure.getTerm(2);
108 				return new JiPrologStructure(provider, left, functor, right);
109 			}
110 		} else if (prologTerm instanceof JIPCons) {
111 			JIPCons cons = (JIPCons) prologTerm;
112 			JIPTerm left = cons.getHead();
113 			JIPTerm right = cons.getTail();
114 			return new JiPrologStructure(provider, left, ",", right);
115 		}
116 		throw new UnknownTermError(prologTerm);
117 	}

119 	public JIPTerm fromTerm(PrologTerm term) {
120 		switch (term.getType()) {
121 		case PrologTermType.NIL_TYPE:
122 			return JIPAtom.create("nil");
123 		case PrologTermType.CUT_TYPE:
124 			return JIPAtom.create("!");
125 		case PrologTermType.FAIL_TYPE:
126 			return JIPAtom.create("fail");
127 		case PrologTermType.TRUE_TYPE:
128 			return JIPAtom.create("true");
129 		case PrologTermType.FALSE_TYPE:
130 			return JIPAtom.create("false");
131 		case PrologTermType.ATOM_TYPE:
132 			return JIPAtom.create(((PrologAtom) term).getStringValue());
133 		case PrologTermType.FLOAT_TYPE:
134 			return JIPNumber.create(((PrologFloat) term).getFloatValue());
135 		case PrologTermType.INTEGER_TYPE:
136 			return JIPNumber.create(((PrologInteger) term).getIntegerValue());
137 		case PrologTermType.DOUBLE_TYPE:
138 			return JIPNumber.create(((PrologDouble) term).getDoubleValue());
139 		case PrologTermType.LONG_TYPE:
140 			return JIPNumber.create(((PrologLong) term).getLongValue());
141 		case PrologTermType.VARIABLE_TYPE:
142 			String name = ((PrologVariable) term).getName();
143 			JIPTerm variable = sharedPrologVariables.get(name);
144 			if (variable == null) {
145 				variable = JIPVariable.create(name);
146 				sharedPrologVariables.put(name, variable);
147 			}
148 			return variable;
149 		case PrologTermType.LIST_TYPE:
150 			PrologTerm[] arguments = ((PrologList) term).getArguments();
151 			JIPList list = JIPList.NIL;
152 			for (int i = arguments.length - 1; i >= 0; --i) {
153 				list = JIPList.create(fromTerm(arguments[i]), list);
154 			}
155 			return list;
156 		case PrologTermType.STRUCTURE_TYPE:
157 			String functor = term.getFunctor();
158 			arguments = ((PrologStructure) term).getArguments();
159 			JIPCons cons = adaptCons(arguments);
160 			return JIPFunctor.create(removeQuoted(functor), cons);
161 		case PrologTermType.OBJECT_TYPE:
162 			return JiPrologReference.set(term.getObject());
163 		default:
164 			throw new UnknownTermError(term);
165 		}
166 	}

168 	public JIPTerm[] fromTermArray(PrologTerm[] terms) {
169 		JIPTerm[] prologTerms = new JIPTerm[terms.length];
170 		for (int i = 0; i < terms.length; i++) {
171 			prologTerms[i] = fromTerm(terms[i]);
172 		}
173 		return prologTerms;
174 	}

176 	public JIPTerm fromTerm(PrologTerm head, PrologTerm[] body) {
177 		JIPTerm clauseHead = fromTerm(head);
178 		if (body != null && body.length > 0) {
179 			JIPTerm clauseBody = fromTerm(body[body.length - 1]);
180 			for (int i = body.length - 2; i >= 0; --i) {
181 				clauseBody = JIPFunctor.create(",", JIPCons.create(fromTerm(body[i]), clauseBody));
182 			}
183 			return JIPFunctor.create(":-", JIPCons.create(clauseHead, clauseBody));
184 		}
185 		return clauseHead;
186 	}

197 	JIPCons adaptCons(PrologTerm[] arguments) {
198 		JIPCons cons = null;
199 		for (int i = arguments.length - 1; i >= 0; --i) {
200 			cons = JIPCons.create(fromTerm(arguments[i]), cons);
201 		}
202 		return cons;
203 	}
