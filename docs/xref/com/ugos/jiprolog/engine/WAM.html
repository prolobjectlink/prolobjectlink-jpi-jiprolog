288     Node backtrack(Node curNode)
289     {
290         Node backtrack;
291         while(curNode != null)
292         {
293 //            System.out.println("redo: " + curNode.getGoal());
294 
295             // risale l'albero saltando i punti di backtracking
296             if(curNode.m_backtrack != null)
297             {
298                 backtrack = curNode.m_backtrack;
299                 do
300                 {
301                     // Azzera le variabili eventualmente istanziate al livello corrente
302                     // poich� in hasMoreElements viene riprovato il match
303                     curNode.clearVariables();
304 
305                     // call precedente
306                     curNode = curNode.m_previous;
307 
308 //                    //aggiorna il backtraking
309                     if(curNode.m_backtrack != null)
310                     {
311                         if(curNode.m_backtrack.m_nLevel < backtrack.m_nLevel)
312                         {
313                             backtrack = curNode.m_backtrack;                                                //
314                         }
315                     }
316                 }
317                 while(backtrack != curNode);
318             }
319 
320             curNode.clearVariables();
321 
322             if(curNode == m_rootNode)
323             {
324                 // non ho altro backtracking
325                 return null;
326             }
327 
328             if(curNode.getGoal() instanceof BuiltInPredicate)
329             {
330                 if(((BuiltInPredicate)curNode.getGoal()).hasMoreChoicePoints())
331                     return curNode;
332             }
333             else if(curNode.m_ruleEnum.hasMoreElements())
334             {
335                 // se la prossima regola unificante fallisce nel corpo qui non
336                 // � possibile accorgersene (il risultato cio� non � deterministico)
337                 // quindi hasMoreElement ritorna true
338                 // occorrerebbe eseguire un look-haed per verificare se la prossima
339                 // regola unifica
340 
341                 return curNode;
342             }
343 
344             curNode = curNode.m_previous;
345         }
346 
347         return null;
348     }

350     boolean run(Node curNode)
351     {
352         PrologRule  rule = null;
353         Clause      clause = null;
354         boolean     bUnify;
355         Hashtable   varTbl;
356         Node        newNode = null;
357         Node        parentNode;
358         int         nCallCount = m_nBaseCounter;
359 
360         try
361         {
362             while(curNode != null)
363             {
364                 // servea a terminare l'esecuzione in seguito alla close
365                 // trovare un metodo migliore
366                 if(m_startNode == null)
367                 {
368 //                    System.out.println("m_startNode == null");
369                 	throw new JIPAbortException();
370                 }
371 
372                 m_curNode = curNode;
373 
374                 // genera le clausole che unificano
375                 // se le clausole sono state gi� generate siamo in backtracking
376                 // altrimenti ne genera di nuove
377                 if(curNode.m_ruleEnum == null)
378                 {
379                     try
380                     {
381                         curNode.m_ruleEnum = getRules(curNode);
382 //                        System.out.println("rule en " + curNode.m_ruleEnum);  // dbg
383                     }
384                     catch(UndefinedPredicateException ex)
385                     {
386                         // invia il warning se il predicato non � definito
387                         // e non � dynamic
388                         // in questo caso la eneration deve essere vuota
389                         if(!m_globalDB.isDynamic(ex.getPredicateName()))
390                         {
391                             ex.m_curNode = curNode;
392                             m_engine.notifyEvent(JIPEvent.ID_UNDEFPREDICATE, Atom.createAtom(ex.getPredicateName()), hashCode());
393                         }
394 
395                         curNode.m_ruleEnum = s_emptyEnum;
396                     }
397                 }
398 
399                 nCallCount++;
400                 curNode.m_nLevel = nCallCount;
401 
402                 bUnify = false;
403                 varTbl = new Hashtable(10); // imposta l'hashtable per le variabili
404                 while(curNode.m_ruleEnum.hasMoreElements() && !bUnify)
405                 {
406                     rule   = (PrologRule)curNode.m_ruleEnum.nextElement();
407                     clause = rule.m_cons;
408                     // UNIFY
409                     // unifica la testa della clausola con il predicato corrente
410                     bUnify = curNode.getGoal().unify(clause.getHead(), varTbl);
411                 }
412 
413                 // verifica la presenza di almeno una clausola
414                 if(bUnify)
415                 {
416 //                    System.out.println("unified " + clause);  // dbg
417 
418                     // imposta l'hashtable delle variabili instanziate nel nodo corrente
419                     curNode.m_varTbl = varTbl;
420 
421                     // FOUND
422                     //System.out.println("curNode call list  " + curNode.m_callList);  // dbg
423                     newNode = null;
424                     if(clause.getTail() != null) // la clausola ha un body
425                     {
426 //                      System.out.println("clause.getTail() != null");  // dbg
427 //                      System.out.println("clause.getTail() " + clause.getTail());  // dbg
428 
429                         // crea un nuovo nodo
430                         newNode = new Node((ConsCell)clause.getTail(), curNode, curNode, rule.m_strModule);
431                         //parentRule = rule;
432                     }
433                     else if(curNode.m_callList.getTail() != null) // la clausola non ha un body continuo con il resto
434                     {
435                         // crea un nuovo nodo
436                         newNode = new Node((ConsCell)curNode.m_callList.getTail(), curNode.m_parent, curNode, curNode.m_strModule);
437                     }
438                     else
439                     {
440                         parentNode = curNode.m_parent;
441                         //System.out.println("parentNode" + parentNode.m_callList);
442 
443                         while(newNode == null && parentNode != null)
444                         {
445                             if(((ConsCell)parentNode.m_callList.getTail()) != null)
446                             {
447                                 newNode = new Node((ConsCell)parentNode.m_callList.getTail(), parentNode.m_parent, curNode, parentNode.m_strModule);
448                             }
449                             else
450                             {
451                                 parentNode = parentNode.m_parent;
452                             }
453                         }
454 
455                         if(newNode == null)
456                         {
457                             m_lastNode = curNode;
458                             m_curNode = null;
459                             return true;
460                         }
461 
462 //                        newNode.m_nLevel = m_curNode.m_nLevel + 1;
463                     }
464 
465                     curNode = newNode;
466                 }
467                 else
468                 {
469                     // FAIL
470                     // non ci sono clausole unificanti
471                     // BACKTRACK
472 //                    notifyFail(curCall);
473 
474 //                    System.out.println("Fail " + curNode.getGoal());  // dbg
475                     curNode.m_ruleEnum = null;
476                     curNode.clearVariables();
477                     curNode = backtrack(curNode.m_previous);
478                 }
479             }
480         }
481         catch(JIPRuntimeException ex)
482         {
483 //            notifyStop();
484             ex.printStackTrace();  //DBG
485 
486             m_curNode = null;
487             m_startNode = null;

495                 if(cla != null)
496                 {
497                     ex.m_strFileName = cla.getFileName(); // nel caso di built-in non � valorizzato
498                     ex.m_nLineNumber = cla.getLineNumber();
499                     ex.m_nPosition = cla.getPosition();

501                 }
502             }
503             throw ex;
504         }

582     final Enumeration getRules(final Node curNode)
583     {
584         PrologObject term = curNode.getGoal();
585 
586         // check if variable (used in metacall variable
587         if (term instanceof Variable)
588         {
589             term = ((Variable)term).getObject();
590 
591             if(term != null)
592             {
593                 curNode.setGoal(term);
594             }
595             else
596             {
597             	throw new JIPParameterUnboundedException();
598 //                throw JIPRuntimeException.create(23, curNode.getGoal());
599             }
600         }
601 
602         // check type
603         if (term instanceof Atom)
604         {
605             term = Functor.getFunctor(term);
606             curNode.setGoal(term);
607         }
608 
609         if (term instanceof BuiltInPredicate)
610         {
611             return new RulesEnumerationBuiltIn((BuiltInPredicate)term, curNode.m_strModule, this);
612         }
613         else if (term instanceof Functor)
614         {
615             String strModule = curNode.m_strModule;
616             // controlla se si tratta di :
617             if(((Functor)term).getName().equals(":/2"))
618             {
619                 strModule = ((Atom)((Functor)term).getParams().getHead()).getName();
620                 term = ((ConsCell)((Functor)term).getParams().getTail()).getHead();
621                 term = Functor.getFunctor(term);
622 
623                 curNode.m_callList.setHead(term);
624             }
625 
626             if (term instanceof BuiltInPredicate)
627                 return new RulesEnumerationBuiltIn((BuiltInPredicate)term, strModule, this);
628             else
629                 return new RulesEnumeration((Functor)term, strModule, m_globalDB);
630         }
631         else if (term instanceof List)
632         {
633             // consult/1
634             term = new BuiltInPredicate("consult/1", new ConsCell(term, null));
635             curNode.setGoal(term);
636             return new RulesEnumerationBuiltIn((BuiltInPredicate)term, curNode.m_strModule, this);
637         }
638         else if(term instanceof ConsCell)
639         {
640             PrologObject head = ((ConsCell)term).getHead();
641             PrologObject tail = ((ConsCell)term).getTail();
642 
643             ConsCell callList = new ConsCell(head, null);
644 
645             while(tail != null)
646             {
647               head = ((ConsCell)tail).getHead();
648 
649               if(head != null)
650                   callList = ConsCell.append(callList, new ConsCell(head, null));
651 
652               tail = ((ConsCell)tail).getTail();
653             }
654 
655             curNode.m_callList = ConsCell.append(callList, (ConsCell)curNode.m_callList.getTail());
656 
657             return getRules(curNode);
658         }
659 
660         throw new JIPParameterTypeException();
661     }
