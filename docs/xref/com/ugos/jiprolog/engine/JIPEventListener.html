21  package com.ugos.jiprolog.engine;

23  /** JIPEventListeners are notified when an interpreter event has occured during execution.<br>
24    * Interpreter events are OPEN, CLOSE, SOLUTION, MORE, END, TERM, ERROR
25    * They are notified in the related methods.
26    * @version 2.0
27    * @author Ugo Chirico<br>
28    * Home Page: http://www.ugochirico.com
29    * @see com.ugos.jiprolog.engine.JIPEvent
30    */
31  public interface JIPEventListener
32  {
33     /** Notifies a Solution event
34      * @param e JIPEvent occurred
35      * @see com.ugos.jiprolog.engine.JIPEvent
36      */
37      public void solutionNotified(JIPEvent e);
38      
39      /** Notifies a Term event. <br>
40       * This event can be generated by notify/2
41       * @param e JIPEvent occurred
42       * @see com.ugos.jiprolog.engine.JIPEvent
43       */
44      public void termNotified(JIPEvent e);
45      
46     /** Notifies an Open event
47      * @param e JIPEvent occurred
48      * @see com.ugos.jiprolog.engine.JIPEvent
49      */
50      public void openNotified(JIPEvent e);
51  
52     /** Notifies a More event
53      * @param e JIPEvent occurred
54      * @see com.ugos.jiprolog.engine.JIPEvent
55      */
56      public void moreNotified(JIPEvent e);
57      
58     /** Notifies an End event
59      * @param e JIPEvent occurred
60      * @see com.ugos.jiprolog.engine.JIPEvent
61      */
62      public void endNotified(JIPEvent e);
63  
64     /** Notifies a Close event
65      * @param e JIPEvent occurred
66      * @see com.ugos.jiprolog.engine.JIPEvent
67      */
68      public void closeNotified(JIPEvent e);
69          
70      /** Receives an error event
71      * @param e JIPErrorEvent object representing the error occurred
72      * @see com.ugos.jiprolog.engine.JIPErrorEvent
73      */
74      public void errorNotified(JIPErrorEvent e);
75  }
