26  final class XCall2 extends BuiltIn
27  {
28      private JIPXCall      m_exObj;
29      private static Hashtable<String, Class> classTable = new Hashtable<String, Class>();
30  
31      // Called by prolog engine when it tries to unify the goal
32      // (in this case the goal is a call to a built in predicate)
33      public final boolean unify(final Hashtable<Variable, Variable> varsTbl)
34      {
35          if (m_exObj == null)  // Called for the first time
36          {
37              // Get JIPXCall class Name (first parameter)
38              final PrologObject exClass = getRealTerm(getParam(1));
39  
40              // extract the Atom related to the class name
41              String strXClassName;
42              if(exClass instanceof PString)
43                  strXClassName = ((PString)exClass).getString();
44              else if(exClass instanceof Atom)
45                  strXClassName = ((Atom)exClass).getName();
46              else
47                  throw new JIPParameterTypeException(1, JIPParameterTypeException.ATOM_OR_STRING);
48  
49              // Create an instance of JIPXCall class
50              m_exObj = createXCall(strXClassName);
51  
52              // Set current JIPEngine instance
53              m_exObj.init(this);
54          }
55  
56          final PrologObject params = getRealTerm(getParam(2));
57              if(!(params instanceof List))
58                  throw new JIPParameterTypeException(2, JIPParameterTypeException.LIST);
59  
60          JIPCons exParams = new JIPCons(((List)params).getConsCell());
61  
62          Hashtable<JIPVariable, JIPVariable> jipVarsTable = new Hashtable<JIPVariable, JIPVariable>();
63          // Invoke JIPXCall class
64          boolean unify = m_exObj.unify(exParams, jipVarsTable);
65  
66          if(unify)
67          {
68          	Variable var;
69          	for(JIPVariable jvar : jipVarsTable.values())
70          	{
71          		var = (Variable)jvar.getTerm();
72          		varsTbl.put(var, var);
73          	}
74          }
75  
76          return unify;
77      }
78  
79  
80      // return true if the JIPXCall class is deterministic
81      public final boolean hasMoreChoicePoints()
82      {
83          return m_exObj == null ? true : m_exObj.hasMoreChoicePoints();
84      }
85  
86      // Create an instance of JIPXCall class
87      @SuppressWarnings("rawtypes")
88  	protected static final JIPXCall createXCall(String strXClassName)
89      {
90          try
91          {
92          	//System.out.println(strXClassName);
93              // Get the correct class name
94              if(strXClassName.charAt(0) == 39 || strXClassName.charAt(0) == 34)
95              {
96                  strXClassName = strXClassName.substring(1, strXClassName.length() - 1);
97              }
98  
99              JIPXCall exObj;
100             Class xclass;
101             if(classTable.contains(strXClassName))
102             {
103             	xclass = classTable.get(strXClassName);
104             }
105             else if(JIPEngine.getClassLoader() != null)
106             {
107             	xclass = JIPEngine.getClassLoader().loadClass(strXClassName);
108             	classTable.put(strXClassName, xclass);
109             }
110             else
111             {
112             	xclass = Class.forName(strXClassName);
113             	classTable.put(strXClassName, xclass);
114             }
115 
116             exObj = (JIPXCall)xclass.newInstance();
117 
118             return exObj;
119         }
120         catch(ClassNotFoundException ex)
121         {
122         	throw JIPExistenceException.createProcedureException(strXClassName);
123         }
124         catch(IllegalAccessException ex)
125         {
126         	throw JIPExistenceException.createProcedureException(strXClassName);
127 //            throw JIPRuntimeException.create(38, strXClassName);
128         }
129         catch(InstantiationException ex)
130         {
131         	throw JIPExistenceException.createProcedureException(strXClassName);
132 //            throw JIPRuntimeException.create(39, strXClassName);
133         }
134         catch(ClassCastException ex)
135         {
136         	throw JIPExistenceException.createProcedureException(strXClassName);
137 //            throw JIPRuntimeException.create(40, strXClassName);
138         }
139         catch(NoClassDefFoundError ex)
140         {
141         	throw JIPExistenceException.createProcedureException(strXClassName);
142         }
143     }
