45      public boolean unify(final Hashtable varsTbl)
46      {
47          final Vector predDefVect = getPredDefVect();
48  
49          final GlobalDB gdb = getJIPEngine().getGlobalDB();
50          for(int i = 0; i < predDefVect.size(); i++)
51          {
52              gdb.multifile((String)predDefVect.elementAt(i));
53          }
54  
55          return true;
56      }
57  
58      protected Vector getPredDefVect()
59      {
60          PrologObject pred = getRealTerm(getParam(1));
61  
62          if(pred instanceof Functor)
63          {
64              pred = new ConsCell(pred, null);
65          }
66  
67          final Vector predDefVect = new Vector();
68  
69          while (pred != null)
70          {
71  
72              try
73              {
74                  String strPredDef;
75                  PrologObject head = getRealTerm(((ConsCell)pred).getHead());
76  
77                  // head deve essere instanza di funtore /2 del tipo name/arity
78                  if(head instanceof Functor)
79              	{
80                  	if (((Functor)head).getName().equals("//2"))
81  	                {
82  	                    ConsCell params = ((Functor )head).getParams();
83  	                    strPredDef = ((Atom)params.getHead()).getName() + "/" + ((ConsCell)params.getTail()).getHead().toString(getJIPEngine());
84  	                    predDefVect.addElement(strPredDef);
85  	                }
86  	                else
87  	                {
88  	                    strPredDef = ((Functor)head).getName();
89  	                    predDefVect.addElement(strPredDef);
90  	                }
91              	}
92                  else
93                  {
94                      throw new JIPParameterTypeException(1, JIPParameterTypeException.PREDICATE_INDICATOR);
95                  }
96  
97                  pred = getRealTerm(((ConsCell)pred).getTail());
98              }
99              catch(ClassCastException ex)
100             {
101                 throw new JIPParameterTypeException(1, JIPParameterTypeException.PREDICATE_INDICATOR);
102             }
103         }
104 
105         return predDefVect;
106     }
