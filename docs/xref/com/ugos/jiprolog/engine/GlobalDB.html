87  		if (isSystem(strPredName) && !m_bCheckDisabled)
88  			throw new JIPPermissionException("modify", "static_procedure", strPredName, jipEngine);
89  
90  		int nPos = strPredName.lastIndexOf('/');
91  		if (nPos < 0)
92  			throw new JIPParameterTypeException(1, JIPParameterTypeException.PREDICATE_INDICATOR, jipEngine);
93  
94  		final String strDef = USER_MODULE + ":" + strPredName;
95  		JIPClausesDatabase db;
96  		if (m_clauseTable.containsKey(strDef)) {
97  			db = ((JIPClausesDatabase) m_clauseTable.get(strDef));
98  		} else {
99  			final String strName = strPredName.substring(0, nPos);
100 			final int nArity = Integer.parseInt(strPredName.substring(nPos + 1));
101 			db = new DefaultClausesDatabase(strName, nArity);
102 			db.setJIPEngine(jipEngine);
103 
104 			// db = new DefaultClausesDatabase();
105 			// Aggiunge il vettore alla tabella
106 			m_clauseTable.put(strDef, db);
107 		}
108 
109 		if (db instanceof DefaultClausesDatabase)
110 			db.setMultifile();
111 		else
112 			throw new JIPPermissionException("modify", "extern_procedure", strPredName);
113 
114 	}

260 	final Clause retract(Clause clause) {
261 		// System.out.println("retract");
262 
263 		Functor functor = (Functor) clause.getHead();
264 
265 		// System.out.println("functor " + functor); // DBG
266 
267 		if (isSystem(functor.getName()))
268 			throw new JIPPermissionException("modify", "static_procedure", functor.getName());
269 		// throw JIPRuntimeException.create(13, functor);
270 
271 		JIPClausesDatabase db;
272 		db = search(functor, clause.getModuleName());
273 
274 		if (db == null)
275 			return null;
276 
277 		Enumeration en = db.clauses();
278 
279 		if (!en.hasMoreElements())
280 			return null;
281 
282 		// System.out.println("more");
283 		boolean bFound = false;
284 		Clause currentClause = null;
285 		while (en.hasMoreElements() && !bFound) {
286 			currentClause = ((Clause) en.nextElement());// .getHead();
287 
288 			if (clause.getTail() == null) {
289 				// si tratta solo di funtore
290 				bFound = functor.unifiable(currentClause.getHead());
291 			} else {
292 				// si tratta di clausola
293 				bFound = clause.unifiable(currentClause);
294 			}
295 		}
296 
297 		if (bFound) {
298 			db.removeClause(new JIPClause(currentClause));
299 			return currentClause;
300 		} else {
301 			return null;
302 		}
303 	}

305 	final void abolish(PrologObject pred) {
306 		if (pred instanceof Variable)
307 			pred = ((Variable) pred).getObject();
308 
309 		if (!(pred instanceof List))
310 			pred = new ConsCell(pred, null);
311 
312 		String strModuleName = USER_MODULE;
313 		PrologObject head = pred;
314 		try {
315 			while (pred != null) {
316 				head = ((ConsCell) pred).getHead();
317 				if (head instanceof Variable)
318 					head = ((Variable) head).getObject();
319 
320 				// controlla se :/2
321 				if (head instanceof Functor && ((Functor) head).getName().equals(":/2")) {
322 					strModuleName = ((Atom) ((Functor) head).getParams().getHead()).getName();
323 					head = ((ConsCell) ((Functor) head).getParams().getTail()).getHead();
324 				}
325 
326 				String strPredDef;
327 				// head deve essere instanza di funtore /2 del tipo name/arity
328 				if (head instanceof Functor && ((Functor) head).getName().equals("//2")) {
329 					ConsCell params = ((Functor) head).getParams();
330 					PrologObject h = params.getHead();
331 					if (h instanceof Variable)
332 						h = ((Variable) h).getObject();
333 					strPredDef = ((Atom) h).getName() + "/" + ((ConsCell) params.getTail()).getHead();
334 				} else {
335 					throw new JIPParameterTypeException(1, JIPParameterTypeException.PREDICATE_INDICATOR);
336 				}
337 
338 				// if(head instanceof Atom) // atom deve essere un preddef
339 				// head = new Functor(((Atom)head).getName(), null);
340 
341 				// if(isSystem(((Functor)head).getName()))
342 				// throw JIPRuntimeException.create(18, head, null);
343 				if (isSystem(strPredDef))
344 					throw new JIPPermissionException("modify", "static_procedure", head.toString());
345 				// throw JIPRuntimeException.create(12, head);
346 
347 				// aggiungere il modulo
348 
349 				// m_clauseTable.remove(strModuleName + ":" +
350 				// ((Functor)head).getName());
351 				m_clauseTable.remove(strModuleName + ":" + strPredDef);
352 
353 				pred = ((ConsCell) pred).getTail();
354 			}
355 		} catch (ClassCastException ex) {
356 			throw new JIPParameterTypeException(1, JIPParameterTypeException.PREDICATE_INDICATOR);
357 		}
358 	}

361 	private final synchronized void addPredicate(final Clause clause, final boolean bFirst, final String strFile, boolean dynamic) {
362 		if (!m_bCheckDisabled && isSystem((Functor) clause.getHead()))
363 			throw new JIPPermissionException("modify", "static_procedure", ((Functor) clause.getHead()).getName(), jipEngine);
364 
365 		// Estrae il nome del funtore
366 		PrologObject head = clause.getHead();
367 
368 		// qui controllare se il funtore � nella export list corrente.
369 		// se si controllare se l'eventuale modulo specificato corrisponde a
370 		// quello di definizione. in tal caso la specifica di modulo va ignorata
371 		String strFunctName = clause.isExported() ? GlobalDB.USER_MODULE : clause.getModuleName();
372 
373 		if (head instanceof Functor) {
374 			strFunctName += ":" + ((Functor) head).getName();
375 		} else {
376 			JIPTypeException ex = new JIPTypeException(JIPTypeException.PREDICATE_INDICATOR, head.toString());
377 			ex.m_engine = jipEngine;
378 			throw ex;
379 		}
380 
381 		// verifica se il predicato � stato gi� asserted in un altro file
382 		if (strFile != null) {
383 			if (m_pred2FileMap.containsKey(strFunctName)) {
384 				String strFileName = (String) m_pred2FileMap.get(strFunctName);
385 				if (!strFileName.equals(strFile) && !isMultifile(((Functor) head).getName())) {
386 					throw new JIPPermissionException("modify", "static_procedure", strFunctName, jipEngine);
387 				}
388 			}
389 		}
390 
391 		JIPClausesDatabase db;
392 
393 		// Verifica l'esistenza del funtore
394 		if (m_clauseTable.containsKey(strFunctName)) {
395 			// System.out.println("database: " + strFunctName);
396 
397 			// Estrae il database
398 			db = (JIPClausesDatabase) (m_clauseTable.get(strFunctName));
399 
400 			if (dynamic && !db.isDynamic()) {
401 				throw new JIPPermissionException("modify", "static_procedure", ((Functor) clause.getHead()).getName(), jipEngine);
402 				// throw JIPRuntimeException.create(30,
403 				// ((Functor)clause.getHead()).getName());
404 			}
405 
406 			if (bFirst) {
407 				// Aggiunge il predicato
408 				if (!db.addClauseAt(0, new JIPClause(clause)))
409 					throw new JIPPermissionException("modify", "static_procedure", ((Functor) clause.getHead()).getName(), jipEngine);
410 				// throw JIPRuntimeException.create(10, clause);
411 			} else {
412 				// Aggiunge il predicato
413 				if (!db.addClause(new JIPClause(clause)))
414 					throw new JIPPermissionException("modify", "static_procedure", ((Functor) clause.getHead()).getName(), jipEngine);
415 				// throw JIPRuntimeException.create(10, clause);
416 			}
417 		} else {
418 			// System.out.println("new database: " + strFunctName);
419 			// Crea un nuovo database
420 			db = new DefaultClausesDatabase(((Functor) head).getFriendlyName(), ((Functor) head).getArity());
421 			db.setJIPEngine(jipEngine);
422 
423 			if (dynamic)
424 				db.setDynamic();
425 
426 			// Aggiunge il predicato
427 			if (!db.addClause(new JIPClause(clause)))
428 				throw new JIPPermissionException("modify", "static_procedure", ((Functor) clause.getHead()).getName(), jipEngine);
429 			// throw JIPRuntimeException.create(10, clause);
430 
431 			// Aggiunge il vettore alla tabella
432 			m_clauseTable.put(strFunctName, db);
433 		}
434 
435 		if (strFile != null) {
436 			m_pred2FileMap.put(strFunctName, strFile);
437 			if (m_file2PredMap.containsKey(strFile)) {
438 				Vector<String> preds = m_file2PredMap.get(strFile);
439 				if (!preds.contains(strFunctName))
440 					preds.add(strFunctName);
441 			} else {
442 				Vector<String> preds = new Vector<String>();
443 				preds.add(strFunctName);
444 				m_file2PredMap.put(strFile, preds);
445 			}
446 		}
447 
448 		if (m_moduleTransparentTbl.containsKey(strFunctName))
449 			db.setModuleTransparent();
450 	}

482 	private static final synchronized void loadKernel(GlobalDB gdb) {
483 		try {
484 			gdb.m_bCheckDisabled = true;
485 
486 			if (JIPDebugger.debug) {
487 				System.out.println("GlobalDebug");
488 
489 				InputStream ins = gdb.getClass().getResourceAsStream(JIPEngine.RESOURCEPATH + "jipkernel.txt");
490 
491 				PrologParser parser = new PrologParser(new ParserReader(new InputStreamReader(ins)), new OperatorManager(), "jipkernel.txt");
492 
493 				PrologObject term;
494 				while ((term = parser.parseNext()) != null) {
495 					// System.out.println(term);
496 					gdb.assertz(Clause.getClause(term), "__KERNEL__", false);
497 				}
498 
499 				ins.close();
500 			} else {
501 				InputStream ins = gdb.getClass().getResourceAsStream(JIPEngine.RESOURCEPATH + "jipkernel.jip");
502 
503 				final ObjectInputStream oins = new ObjectInputStream(ins);
504 				ArrayList<PrologObject> program = (ArrayList<PrologObject>) oins.readObject();
505 				oins.close();
506 				ins.close();
507 
508 				for (PrologObject term : program) {
509 					// System.out.println(term);
510 					gdb.assertz(Clause.getClause(term), "__KERNEL__", false);
511 				}
512 			}
513 
514 			gdb.moduleTransparent("\\+/1");
515 			gdb.moduleTransparent("not/1");
516 			gdb.moduleTransparent(";/2");
517 			gdb.moduleTransparent("->/2");
518 			gdb.moduleTransparent("^/2");
519 
520 			gdb.m_bCheckDisabled = false;
521 		} catch (IOException ex) {
522 			gdb.m_bCheckDisabled = false;
523 			throw new JIPRuntimeException("Unable to load Kernel: " + ex.toString());
524 		} catch (JIPSyntaxErrorException ex) {
525 			gdb.m_bCheckDisabled = false;
526 			throw new JIPRuntimeException("Unable to load Kernel: " + ex.toString());
527 		} catch (NullPointerException ex) {
528 			// ex.printStackTrace();
529 			gdb.m_bCheckDisabled = false;
530 			throw new JIPRuntimeException("JIProlog Kernel is not found. JIPKernel.txt should be in the classpath.");
531 		} catch (ClassNotFoundException e) {
532 			// ex.printStackTrace();
533 			gdb.m_bCheckDisabled = false;
534 			throw new JIPRuntimeException("Unable to load Kernel: " + e.toString());
535 		}
536 	}

561 	private final PrologObject fixTerm(final PrologObject term) {
562 		// System.out.println("fixterm");
563 		// System.out.println(term);
564 
565 		if (term instanceof BuiltInPredicate) {
566 			return new BuiltInPredicate(((Functor) term).getAtom(), (ConsCell) fixTerm(((BuiltInPredicate) term).getParams()));
567 		} else if (term instanceof Functor) {
568 			return new Functor(((Functor) term).getAtom(), (ConsCell) fixTerm(((Functor) term).getParams()));
569 		} else if (term instanceof PString) {
570 			// System.out.println("ConsCell");
571 
572 			return new PString(new List(fixTerm(((ConsCell) term).getHead()), fixTerm(((ConsCell) term).getTail())));
573 		} else if (term instanceof List) {
574 			// System.out.println("ConsCell");
575 
576 			if (term == List.NIL)
577 				return term;
578 			else
579 				return new List(fixTerm(((ConsCell) term).getHead()), fixTerm(((ConsCell) term).getTail()));
580 		} else if (term instanceof Clause) {
581 			// System.out.println("Clause");
582 			Clause clause = new Clause(((Clause) term).getModuleName(), (Functor) fixTerm(((ConsCell) term).getHead()), (ConsCell) fixTerm(((ConsCell) term).getTail()));
583 			if (((Clause) term).isExported() || isExported((Functor) clause.getHead()))
584 				clause.setExported();
585 
586 			clause.setFileName(((Clause) term).getFileName());
587 			clause.setLineNumber(((Clause) term).getLineNumber());
588 			clause.setPosition(((Clause) term).getPosition());
589 
590 			return clause;
591 		} else if (term instanceof ConsCell) {
592 			if (term == ConsCell.NIL)
593 				return term;
594 			else
595 				return new ConsCell(fixTerm(((ConsCell) term).getHead()), fixTerm(((ConsCell) term).getTail()));
596 		} else if (term instanceof Variable) {
597 			// return term;
598 			if (((Variable) term).isBounded()) {
599 				return fixTerm(((Variable) term).getObject());
600 			} else {
601 				return ((Variable) term).lastVariable();
602 			}
603 		}
604 
605 		return term;
