23  /**
24   * JIPTraceEvent is used to notify a thar a Trace event occured.
25   * Note that trace events are raised only when trace flag is set to true.<br>
26   * JIPTraceListeners receive a JIPTraceEvent object in one of their methods depending on the type of event occurred
27   * @version 2.0
28   * @author Ugo Chirico 2002<br>
29   * Home Page: http://www.ugochirico.com
30   * @see com.ugos.jiprolog.engine.JIPTraceListener
31   * @see com.ugos.jiprolog.engine.JIPTraceListener#startNotified
32   * @see com.ugos.jiprolog.engine.JIPTraceListener#callNotified
33   * @see com.ugos.jiprolog.engine.JIPTraceListener#foundNotified
34   * @see com.ugos.jiprolog.engine.JIPTraceListener#bindNotified
35   * @see com.ugos.jiprolog.engine.JIPTraceListener#exitNotified
36   * @see com.ugos.jiprolog.engine.JIPTraceListener#failNotified
37   * @see com.ugos.jiprolog.engine.JIPTraceListener#redoNotified
38   * @see com.ugos.jiprolog.engine.JIPTraceListener#stopNotified
39   */
40  public class JIPTraceEvent extends JIPEvent
41  {
42      private WAMTrace m_wam;
43  
44      /**
45      * ID of a Call event
46      */
47      public static final int ID_CALL  = -20;
48  
49      /**
50      * ID of a Bind event
51      */
52      public static final int ID_BIND = -21;
53      
54      /**
55      * ID of a Found event
56      */
57      public static final int ID_FOUND = -22;
58      
59      /**
60      * ID of a Redo event
61      */
62      public static final int ID_REDO  = -23;
63  
64      /**
65      * ID of a Fail event
66      */
67      public static final int ID_FAIL  = -24;
68      
69      /**
70      * ID of a Exit event
71      */
72      public static final int ID_EXIT  = -25;
73      
74      /**
75      * ID of a Start event
76      */
77      public static final int ID_START  = -26;
78  
79      /**
80      * ID of a Stop event
81      */
82      public static final int ID_STOP  = -27;
83      
84      
85      private static final int ABORT   = 1;
86      private static final int NEXT    = 2;
87      private static final int SKIP    = 3;
88      private static final int RETRY   = 4;
89      
90      private int m_nToDo;
91      private int m_nLevel;
92      
93      JIPTraceEvent(final int nID, final PrologObject term, final JIPEngine source, final int nQueryHandle, final WAMTrace wam, final int nLevel)
94      {
95          super(nID, term, source, nQueryHandle);
96          m_wam = wam;
97          m_nLevel = nLevel;
98      }
99  
100    /**
101     * Gets the trace level
102     */
103     public int getLevel()
104     {
105         return m_nLevel;
106     }
107 
108    /**
109     * Aborts the execution
110     */
111     public void abort()
112     {
113        m_nToDo = ABORT;
114        go();
115     }
116 
117    /**
118     * Executes next step
119     */
120     public void nextStep()
121     {
122        m_nToDo = NEXT;
123        go();
124     }
125     
126     /**
127     * Retry current call
128     */
129     public void retry()
130     {
131        m_nToDo = RETRY;
132        go();
133     }
134     
135    /**
136     * Skips the trace for next call
137     */
138     public void skip()
139     {
140         m_nToDo = SKIP;
141         go();
142     }
143     
144     boolean executionAborted()
145     {
146         return m_nToDo == ABORT;
147     }
148     
149     boolean retryCall()
150     {
151         return m_nToDo == RETRY;
152     }
153     
154     boolean skipped()
155     {
156         return m_nToDo == SKIP;
157     }
158     
159     private void go()
160     {
161         m_wam.notifyUserInput();
162     }
