78      /** Gets the eerror term associated to this exception
79       * @return JIPTerm object associated to this exception
80       */
81      public JIPTerm getTerm()
82      {
83          String strTerm = ((m_term != null) ? (m_term.toString()) : ((m_curNode == null) ? ("undefined") : (m_curNode.getGoal().toString())));
84  
85          if(exceptionTerm != null)
86          {
87          	return getTerm(exceptionTerm.toString(), strTerm);
88          }
89          else
90          {
91          	String strMessage = super.getMessage();
92          	if(strMessage == null) strMessage = "undefined";
93  
94          	strMessage = Atom.createAtom(strMessage).toString();
95  
96          	return getTerm("system_error(" + strMessage + ")", strTerm);
97          }
98      }

100     protected JIPTerm getTerm(final String strType, final String strTerm)
101     {
102         if(m_engine != null)
103         {
104         	if(m_strFileName == null)
105         	    m_strFileName = "undefined";
106 
107         	String strFileName = Atom.createAtom(m_strFileName).toString();
108 
109             try
110             {
111                 return
112                 	m_engine.getTermParser().parseTerm("error(" + Atom.createAtom(strType) + ", context(" + strTerm + ", file(" + strFileName + ", " + m_nLineNumber + ")))");
113 	        }
114 	        catch (JIPSyntaxErrorException ex)
115 	        {
116 	             return m_engine.getTermParser().parseTerm("error(" + strType + ", context(undefined, file(undefined, undefined)))");
117 	        }
118         }
119         else
120         {
121             return ((m_term != null) ? (new JIPTerm(m_term.copy(true))) : ((m_curNode == null) ? (null) : (new JIPTerm(m_curNode.getGoal().copy(true)))));
122         }
123     }

184     /** Write the stack trace of goals to the given print writer
185      * @param pw printwriter that receive the stack trace
186      */
187     public void printPrologStackTrace(PrintWriter pw)
188     {
189         if(m_curNode == null)
190         {
191             pw.println("No stack trace available");
192             pw.flush();
193             return;
194         }
195 
196         WAM.Node node = m_curNode;
197         pw.println(node.getGoal().toString(m_engine));
198         node = node.m_previous;
199 
200         while(node != null)
201         {
202             //System.out.println("node.getGoal() " + node.getGoal());
203             if(node.getGoal() != null)
204                 pw.println(" - " + node.getGoal().toString(m_engine));
205             pw.flush();
206             node = node.m_previous;
207         }
208     }
