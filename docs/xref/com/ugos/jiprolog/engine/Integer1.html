25  {
26      public final boolean unify(final Hashtable varsTbl)
27      {
28          try

30          	PrologObject term = getRealTerm(getParam(1));
31          	if(!(term instanceof Expression))
32          		return false;
33

35          }
36          catch(JIPRuntimeException ex)

38              return false;
39          }
40      }
