28   * @see com.ugos.jiprolog.engine.JIPTerm

30  public class JIPAtom extends JIPTerm implements Clearable
31  {
32      private final static long serialVersionUID = 300000001L;
33      
34      /** Creates a new JIPAtom object from a string
35       * @param strName the atom as string
36       * @return a new JIPAtom object from a string
37       */
38      public static final JIPAtom create(final String strName)
39      {
40          return new JIPAtom(Atom.createAtom(strName));
41      }
42      
43      JIPAtom(final Atom atom)
44      {
45          super(atom);
46      }
47      
48      /** Returns a string rapresentation of this atom
49        */
50      public final String getName()
51      {
52          return ((Atom)getTerm()).getName();
53      }
54  }
