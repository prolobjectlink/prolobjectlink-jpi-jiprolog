23  /**
24   * JIPSyntaxErrorException is an exception raised by the internal parser
25   * when there is a syntax error in the string to parse.<br>
26   * @version 3.0
27   * @author Ugo Chirico 2004<br>
28   * Home Page : http://www.ugochirico.com
29   * @see com.ugos.jiprolog.engine.JIPEngine
30   */
31  public class JIPSyntaxErrorException extends JIPRuntimeException
32  {
33      private String m_strFile;
34      private int m_nLine;
35      private String m_strTerm;
36  
37      JIPSyntaxErrorException(final String strFileName, final int nLineNumber, final String strTerm)
38      {
39          super(createRuntimeException(1, "File: "+ strFileName + ", Line: " + nLineNumber + ", Cause: " + strTerm));
40          m_strFile = strFileName;
41          m_nLine = nLineNumber;
42          m_strTerm = strTerm;
43      }
44  
45  //    JIPSyntaxErrorException(final String msg)
46  //    {
47  //        super(create(1, msg));
48  //    }
49  
50      /** Gets the error term associated to this exception
51       * @return JIPTerm object associated to this exception
52       */
53      public JIPTerm getTerm()
54      {
55          if(m_engine != null)
56          {
57              String strFile = Atom.createAtom(m_strFile).toString();
58  	        String strTerm = Atom.createAtom(m_strTerm).toString();
59  	        try
60  	        {
61  	            return m_engine.getTermParser().parseTerm("error(syntax_error(" + strTerm + "), file(" + strFile + ", " + m_nLine + "))");
62  	        }
63  	        catch (JIPSyntaxErrorException ex)
64  	        {
65  	             return m_engine.getTermParser().parseTerm("error(syntax_error(undefined), file(undefined, undefined))");
66  	        }
67          }
68          else
69          {
70              return super.getTerm();
71          }
72      }
73  }
