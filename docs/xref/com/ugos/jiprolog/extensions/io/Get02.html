24  import java.io.*;
25  import java.util.*;
26  
27  public final class Get02 extends JIPXCall
28  {
29      private String m_strStreamHandle;
30  
31      protected final int readNextChar(InputStream ins)
32      {
33          try
34          {
35              return ins.read();
36          }
37          catch(IOException ex)
38          {
39              throw new JIPJVMException(ex);
40          }
41      }
42  
43      public final boolean unify(final JIPCons params, Hashtable varsTbl)
44      {
45          // get first parameter
46          JIPTerm input = params.getNth(1);
47  
48          // check if input is a variable
49          if (input instanceof JIPVariable)
50          {
51              // try to extract the term
52              if(!((JIPVariable)input).isBounded())
53              {
54                  throw new JIPParameterUnboundedException(1);
55              }
56              else
57              {
58                  //extracts the term
59                  input = ((JIPVariable)input).getValue();
60              }
61          }
62  
63          // check if input is an Atom
64          if(input instanceof JIPAtom)
65          {
66              // Gets the handle to the stream
67              m_strStreamHandle = ((JIPAtom)input).getName();
68  
69              // Get the stream
70              final InputStream ins = JIPio.getInputStream(m_strStreamHandle, getJIPEngine());
71              if(ins == null)
72              {
73              	throw JIPExistenceException.createStreamException(m_strStreamHandle);
74  //            	throw new JIPDomainException("stream_or_alias", m_strStreamHandle);
75              }
76  
77              if("user_input".equals(m_strStreamHandle))
78                  getJIPEngine().notifyEvent(JIPEvent.ID_WAITFORUSERINPUT, getPredicate(), getQueryHandle());
79  
80              JIPTerm term;
81              int c = readNextChar(ins);
82              if(c == -1)
83              {
84              	StreamInfo streamInfo = JIPio.getStreamInfo(m_strStreamHandle);
85              	streamInfo.setEndOfStream("past");
86              }
87  
88  //            else
89                  term = JIPNumber.create(c);
90  
91  //                System.out.println("get0 " + c);
92  
93              if("user_input".equals(m_strStreamHandle))
94                  getJIPEngine().notifyEvent(JIPEvent.ID_USERINPUTDONE, getPredicate(), getQueryHandle());
95  
96              return params.getNth(2).unify(term, varsTbl);
97          }
98          else
99              throw new JIPParameterTypeException(2, JIPParameterTypeException.ATOM);
100     }

103     {
104         return false;
